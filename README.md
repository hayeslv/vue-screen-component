# screen-component
大屏组件库

## 安装

```bash
npm i hay-ui@latest
```

## 使用

```ts
// main.ts
import { createApp } from "vue";
import App from "./App.vue";

import "hay-ui/dist/index.min.css";

const app = createApp(App);
app.mount("#app");
```

```ts
// component
import { HayChart, HayPanelBac } from "hay-ui";
```
